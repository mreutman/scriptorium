\newcommand{\psalmChapter}[1] {%
  \smallskip
  \needspace{3\baselineskip}
  \begin{flushleft}
  {\Alegreya\addfontfeature{LetterSpace=20.0}
    \uppercase{
      {\bfseries\fontdimen2\font=0.8em PSALMUS #1}
    }
  }
  \end{flushleft}
}

\newcommand{\psalmChapterWithInscription}[2] {%
  \smallskip
  \needspace{4\baselineskip}
  \begin{flushleft}
  {\Alegreya\addfontfeature{LetterSpace=20.0}
    \uppercase{
      {\bfseries\fontdimen2\font=0.8em PSALMUS #1}
    }
  }

  {\AlegreyaSC\addfontfeature{LetterSpace=10.0}
    \lowercase{
      {\itshape\small\fontdimen2\font=0.5em #2}
    }
  }
  \end{flushleft}
}
